194 template <class SCALAR>
195 typename SDMM<SCALAR>::Diagnostic SDMM<SCALAR>::operator()(t_Vector &out,
196  t_Vector const &input) const {
197  sanity_check(input);
198  bool convergence = false;
199  t_uint niters(0);
200  // Figures out where itermax or convergence reached
201  auto const has_finished = [&convergence, &niters, this](t_Vector const &out) {
202  convergence = is_converged(out);
203  return niters >= itermax() or convergence;
204  };
205 
206  SOPT_HIGH_LOG("Performing SDMM ");
207  out = input;
208  t_Vectors y(transforms().size()), z(transforms().size());
209 
210  // Initial step replaces iteration update with initialization
211  initialization(y, z, input);
212  auto cg_diagnostic = solve_for_xn(out, y, z);
213 
214  while (not has_finished(out)) {
215  SOPT_LOW_LOG("Iteration {}/{}. ", niters, itermax());
216  // computes y and z from out and transforms
217  update_directions(y, z, out);
218  SOPT_LOW_LOG(" - sum z_ij = {}",
219  std::accumulate(z.begin(), z.end(), Scalar(0e0),
220  [](Scalar const &a, t_Vector const &z) { return a + z.sum(); }));
221  // computes x = L^-1 y
222  cg_diagnostic = solve_for_xn(out, y, z);
223  SOPT_LOW_LOG(" - CG Residual = {} in {}/{} iterations", cg_diagnostic.residual,
224  cg_diagnostic.niters, conjugate_gradient().itermax());
225 
226  ++niters;
227  }
228  return {niters, convergence, cg_diagnostic};
229 }

231 template <class SCALAR>
232 ConjugateGradient::Diagnostic SDMM<SCALAR>::solve_for_xn(t_Vector &out, t_Vectors const &y,
233  t_Vectors const &z) const {
234  assert(z.size() == transforms().size());
235  assert(y.size() == transforms().size());
236  SOPT_TRACE("Solving for x_n");
237 
238  // Initialize b of A x = b = sum_i L_i^H(z_i - y_i)
239  t_Vector b = out.Zero(out.size());
240  for (t_uint i(0); i < transforms().size(); ++i) b += transforms(i).adjoint() * (y[i] - z[i]);
241  if (b.stableNorm() < 1e-12) {
242  out.fill(0e0);
243  return {0, 0, true};
244  }
245 
246  // Then create operator A
247  auto A = [this](t_Vector &out, t_Vector const &input) {
248  out = out.Zero(input.size());
249  for (auto const &transform : this->transforms())
250  out += transform.adjoint() * static_cast<t_Vector>(transform * input);
251  };
252 
253  // Call conjugate gradient
254  auto const diagnostic = this->conjugate_gradient(out, A, b);
255  if (not diagnostic.good) {
256  SOPT_ERROR("CG error - iterations: {}/{} - residuals {}\n", diagnostic.niters,
257  conjugate_gradient().itermax(), diagnostic.residual);
258  SOPT_THROW("Conjugate gradient failed to converge");
259  }
260 
261  return diagnostic;
262 }
