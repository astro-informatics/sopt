195 template <class SCALAR>
196 typename SDMM<SCALAR>::Diagnostic SDMM<SCALAR>::operator()(t_Vector &out,
197  t_Vector const &input) const {
198  sanity_check(input);
199  bool convergence = false;
200  t_uint niters(0);
201  // Figures out where itermax or convergence reached
202  auto const has_finished = [&convergence, &niters, this](t_Vector const &out) {
203  convergence = is_converged(out);
204  return niters >= itermax() or convergence;
205  };
206 
207  SOPT_HIGH_LOG("Performing SDMM ");
208  out = input;
209  t_Vectors y(transforms().size()), z(transforms().size());
210 
211  // Initial step replaces iteration update with initialization
212  initialization(y, z, input);
213  auto cg_diagnostic = solve_for_xn(out, y, z);
214 
215  while (not has_finished(out)) {
216  SOPT_LOW_LOG("Iteration {}/{}. ", niters, itermax());
217  // computes y and z from out and transforms
218  update_directions(y, z, out);
219  SOPT_LOW_LOG(" - sum z_ij = {}",
220  std::accumulate(z.begin(), z.end(), Scalar(0e0),
221  [](Scalar const &a, t_Vector const &z) { return a + z.sum(); }));
222  // computes x = L^-1 y
223  cg_diagnostic = solve_for_xn(out, y, z);
224  SOPT_LOW_LOG(" - CG Residual = {} in {}/{} iterations", cg_diagnostic.residual,
225  cg_diagnostic.niters, conjugate_gradient().itermax());
226 
227  ++niters;
228  }
229  return {niters, convergence, cg_diagnostic};
230 }

232 template <class SCALAR>
233 ConjugateGradient::Diagnostic SDMM<SCALAR>::solve_for_xn(t_Vector &out, t_Vectors const &y,
234  t_Vectors const &z) const {
235  assert(z.size() == transforms().size());
236  assert(y.size() == transforms().size());
237  SOPT_TRACE("Solving for x_n");
238 
239  // Initialize b of A x = b = sum_i L_i^H(z_i - y_i)
240  t_Vector b = out.Zero(out.size());
241  for (t_uint i(0); i < transforms().size(); ++i) b += transforms(i).adjoint() * (y[i] - z[i]);
242  if (b.stableNorm() < 1e-12) {
243  out.fill(0e0);
244  return {0, 0, true};
245  }
246 
247  // Then create operator A
248  auto A = [this](t_Vector &out, t_Vector const &input) {
249  out = out.Zero(input.size());
250  for (auto const &transform : this->transforms())
251  out += transform.adjoint() * (transform * input).eval();
252  };
253 
254  // Call conjugate gradient
255  auto const diagnostic = this->conjugate_gradient(out, A, b);
256  if (not diagnostic.good) {
257  SOPT_ERROR("CG error - iterations: {}/{} - residuals {}\n", diagnostic.niters,
258  conjugate_gradient().itermax(), diagnostic.residual);
259  SOPT_THROW("Conjugate gradient failed to converge");
260  }
261 
262  return diagnostic;
263 }

299  if (r != x.size()) {

301  doexit = true;
