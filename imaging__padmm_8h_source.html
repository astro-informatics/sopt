285 template <class SCALAR>
286 typename ImagingProximalADMM<SCALAR>::Diagnostic ImagingProximalADMM<SCALAR>::operator()(
287  t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
288  SOPT_HIGH_LOG("Performing Proximal ADMM with L1 and L2 operators");
289  // The f proximal is an L1 proximal that stores some diagnostic result
290  Diagnostic result;
291  auto const f_proximal = [this, &result](t_Vector &out, Real gamma, t_Vector const &x) {
292  result.l1_diagnostic = this->l1_proximal(out, gamma, x);
293  };
294  auto const g_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
295  this->l2ball_proximal()(out, gamma, x);
296  };
297  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
298  "Objective function");
299  auto const convergence = [this, scalvar](t_Vector const &x, t_Vector const &residual) mutable {
300  return this->is_converged(scalvar, x, residual);
301  };
302  auto const padmm = PADMM(f_proximal, g_proximal, target())
303  .itermax(itermax())
304  .gamma(gamma())
305  .nu(nu())
306  .lagrange_update_scale(lagrange_update_scale())
307  .Phi(Phi())
308  .is_converged(convergence);
309  static_cast<typename PADMM::Diagnostic &>(result) = padmm(out, std::tie(guess, res));
310  return result;
311 }

313 template <class SCALAR>
314 bool ImagingProximalADMM<SCALAR>::residual_convergence(t_Vector const &x,
315  t_Vector const &residual) const {
316  if (static_cast<bool>(residual_convergence())) return residual_convergence()(x, residual);
317  if (residual_tolerance() <= 0e0) return true;
318  auto const residual_norm = sopt::l2_norm(residual, l2ball_proximal_weights());
319  SOPT_LOW_LOG(" - [PADMM] Residuals: {} <? {}", residual_norm, residual_tolerance());
320  return residual_norm < residual_tolerance();

324 bool ImagingProximalADMM<SCALAR>::objective_convergence(ScalarRelativeVariation<Scalar> &scalvar,
325  t_Vector const &x,

333 template <class SCALAR>
334 bool ImagingProximalADMM<SCALAR>::is_converged(ScalarRelativeVariation<Scalar> &scalvar,
335  t_Vector const &x, t_Vector const &residual) const {
336  auto const user = static_cast<bool>(is_converged()) == false or is_converged()(x, residual);
337  auto const res = residual_convergence(x, residual);
338  auto const obj = objective_convergence(scalvar, x, residual);
339  // beware of short-circuiting!
340  // better evaluate each convergence function everytime, especially with mpi
341  return user and res and obj;
342 }
