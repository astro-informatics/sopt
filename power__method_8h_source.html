21 template <class T>
22 std::tuple<t_real, T> power_method(const sopt::LinearTransform<T> &op, const t_uint niters,
23  const t_real relative_difference, const T &initial_vector) {
24  /*
25  Attempt at coding the power method, returns the sqrt of the largest eigen value of a linear
26  operator composed with its adjoint niters:: max number of iterations relative_difference::
27  percentage difference at which eigen value has converged
28  */
29  if (niters <= 0) return std::make_tuple(1., initial_vector);
30  t_real estimate_eigen_value = 1;
31  t_real old_value = 0;
32  T estimate_eigen_vector = initial_vector;
33  estimate_eigen_vector = estimate_eigen_vector / estimate_eigen_vector.matrix().stableNorm();
34  SOPT_DEBUG("Starting power method");
35  SOPT_DEBUG(" -[PM] Iteration: 0, norm = {}", estimate_eigen_value);
36  bool converged = false;
37  ScalarRelativeVariation<t_real> scalvar(relative_difference, 0., "Eigenvalue");
38  for (t_int i = 0; i < niters; ++i) {
39  estimate_eigen_vector = op.adjoint() * static_cast<T>(op * estimate_eigen_vector);
40  estimate_eigen_value = estimate_eigen_vector.matrix().stableNorm();
41  if (estimate_eigen_value <= 0) throw std::runtime_error("Error in operator.");
42  if (estimate_eigen_value != estimate_eigen_value)
43  throw std::runtime_error("Error in operator or data corrupted.");
44  estimate_eigen_vector = estimate_eigen_vector / estimate_eigen_value;
45  SOPT_DEBUG(" -[PM] Iteration: {}, norm = {}", i + 1, estimate_eigen_value);
46  converged = scalvar(std::sqrt(estimate_eigen_value));
47  old_value = estimate_eigen_value;
48  if (converged) {
49  SOPT_DEBUG("Converged to norm = {}, relative difference < {}", std::sqrt(old_value),
50  relative_difference);
51  break;
52  }
53  }
54  return std::make_tuple(std::sqrt(old_value), estimate_eigen_vector);
55 }

216 typename PowerMethod<SCALAR>::DiagnosticAndResult PowerMethod<SCALAR>::operator()(
217  OperatorFunction<t_Vector> const &op, t_Vector const &input) const {
218  SOPT_INFO("Computing the upper bound of a given operator");
219  SOPT_INFO(" - input vector {}", input.transpose());
220  t_Vector eigenvector = input.normalized();
221  SOPT_INFO(" - eigenvector norm {}", eigenvector.stableNorm());
222  typename t_Vector::Scalar previous_magnitude = 1;
223  bool converged = false;
224  t_uint niters = 0;
225 
226  for (; niters < itermax() and converged == false; ++niters) {
227  op(eigenvector, eigenvector);
228  typename t_Vector::Scalar const magnitude =
229  eigenvector.stableNorm() / static_cast<Real>(eigenvector.size());
230  auto const rel_val = std::abs((magnitude - previous_magnitude) / previous_magnitude);
231  converged = rel_val < tolerance();
232  SOPT_INFO(" - [PM] iteration {}/{} -- norm: {}", niters, itermax(), magnitude);
233 
234  eigenvector /= magnitude;
235  previous_magnitude = magnitude;
236  }
237  // check function exists, otherwise, don't know if convergence is meaningful
238  if (not converged) {
239  SOPT_WARN(" - [PM] did not converge within {} iterations", itermax());
240  } else {
241  SOPT_INFO(" - [PM] converged in {} of {} iterations", niters, itermax());
242  }
243  return DiagnosticAndResult{itermax(), converged, previous_magnitude, eigenvector.normalized()};
