11 sopt::Matrix<T> concatenated_permutations(sopt::t_uint i, sopt::t_uint j) {
12  extern std::unique_ptr<std::mt19937_64> mersenne;
13  std::vector<size_t> cols(j);
14  std::iota(cols.begin(), cols.end(), 0);
15  std::shuffle(cols.begin(), cols.end(), *mersenne);
16 
17  assert(j % i == 0);
18  auto const N = j / i;
19  auto const elem = 1e0 / std::sqrt(static_cast<typename sopt::real_type<T>::type>(N));
20  sopt::Matrix<T> result = sopt::Matrix<T>::Zero(i, cols.size());
21  for (typename sopt::Matrix<T>::Index k(0); k < result.cols(); ++k) result(cols[k] / N, k) = elem;
22  return result;
23 }
