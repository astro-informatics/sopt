99  Eigen::DenseBase<T1> const &x) {
100  typename Eigen::Index const size = x.size() / 2;
101  typename T1::PlainObject const &u = x.segment(0, size);
102  typename T1::PlainObject const &v = x.segment(size, size);
103  out = T0::Zero(size * 2);
104  for (typename Eigen::Index i(0); i < size; i++) {

106  if (norm > gamma) {

108  out(size + i) = (1 - gamma / norm) * v(i);
109  } else {
110  out(i) = 0;
111  out(size + i) = 0;
112  }
