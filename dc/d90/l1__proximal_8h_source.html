126 typename std::enable_if<is_complex<SCALAR>::value == is_complex<typename T0::Scalar>::value and
127  is_complex<SCALAR>::value == is_complex<typename T1::Scalar>::value>::type
128 L1TightFrame<SCALAR>::operator()(Eigen::MatrixBase<T0> &out, Real gamma,
129  Eigen::MatrixBase<T1> const &x) const {
130  Vector<Scalar> const psit_x = Psi().adjoint() * x;
131  if (weights().size() == 1)
132  out = Psi() * (soft_threshhold(psit_x, nu() * gamma * weights()(0)) - psit_x) / nu() + x;
133  else
134  out = Psi() * (soft_threshhold(psit_x, nu() * gamma * weights()) - psit_x) / nu() + x;
135  SOPT_LOW_LOG("Prox L1: objective = {}", objective(x, out, gamma));
136 }

325 template <class SCALAR>
326 template <class T0, class MIXING>
327 typename L1<SCALAR>::Diagnostic L1<SCALAR>::operator()(Eigen::MatrixBase<T0> &out, Real gamma,
328  Vector<Scalar> const &x,
329  MIXING mixing) const {
330  SOPT_MEDIUM_LOG("Starting Proximal L1 operator:");
331  t_uint niters = 0;
332  out = x;
333 
334  Breaker breaker(objective(out, x, gamma), tolerance(), false); // not fista_mixing());
335  SOPT_LOW_LOG(" - [ProxL1] iter {}, prox_fval = {}", niters, breaker.current());
336  Vector<Scalar> const res = Psi().adjoint() * out;
337  Vector<Scalar> u_l1 = 1e0 / nu() * (res - apply_soft_threshhold(gamma, res));
338  apply_constraints(out, x - Psi() * u_l1);
339 
340  // Move on to other iterations
341  for (++niters; niters < itermax() or itermax() == 0; ++niters) {
342  auto const do_break = breaker(objective(x, out, gamma));
343  SOPT_LOW_LOG(" - [ProxL1] iter {}, prox_fval = {}, rel_fval = {}", niters, breaker.current(),
344  breaker.relative_variation());
345  if (do_break) break;
346 
347  Vector<Scalar> const res = u_l1 * nu() + Psi().adjoint() * out;
348  mixing(u_l1, 1e0 / nu() * (res - apply_soft_threshhold(gamma, res)), niters);
349  apply_constraints(out, x - Psi() * u_l1);
350  }
351 
352  if (breaker.two_cycle()) SOPT_WARN("Two-cycle detected when computing L1");
353 
354  if (breaker.converged()) {
355  SOPT_LOW_LOG("Proximal L1 operator converged at {} in {} iterations", breaker.current(),
356  niters);
357  } else
358  SOPT_ERROR("Proximal L1 operator did not converge after {} iterations", niters);
359  return {niters, breaker.relative_variation(), breaker.current(), breaker.converged()};
360 }

384 class L1<SCALAR>::FistaMixing {
385  public:
386  typedef typename real_type<SCALAR>::type Real;
387  FistaMixing() : t(1){};
388  template <class T1>
389  void operator()(Vector<SCALAR> &previous, Eigen::MatrixBase<T1> const &unmixed, t_uint iter) {
390  // reset
391  if (iter == 0) {
392  previous = unmixed;
393  return;
394  }
395  if (iter <= 1) t = next(1);
396  auto const prior_t = t;
397  t = next(t);
398  auto const alpha = (prior_t - 1) / t;
399  previous = (1e0 + alpha) * unmixed.derived() - alpha * previous;
400  }
401  static Real next(Real t) { return 0.5 + 0.5 * std::sqrt(1e0 + 4e0 * t * t); }
402 
403  private:
404  Real t;
405 };
