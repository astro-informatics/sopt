220 template <class SCALAR>
221 void ProximalADMM<SCALAR>::iteration_step(t_Vector &out, t_Vector &residual, t_Vector &lambda,
222  t_Vector &z) const {
223  g_proximal(z, gamma(), -lambda - residual);
224  f_proximal(out, gamma() / nu(), out - Phi().adjoint() * (residual + lambda + z) / nu());
225  residual = Phi() * out - target();
226  lambda += lagrange_update_scale() * (residual + z);
227 }

230 typename ProximalADMM<SCALAR>::Diagnostic ProximalADMM<SCALAR>::
231 operator()(t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
232 
233  SOPT_HIGH_LOG("Performing Proximal ADMM");
234  sanity_check(x_guess, res_guess);
235 
236  t_Vector lambda = t_Vector::Zero(target().size());
237  t_Vector z = t_Vector::Zero(target().size());
238  t_Vector residual = res_guess;
239  out = x_guess;
240 
241  t_uint niters(0);
242  bool converged = false;
243  for(; (not converged) && (niters < itermax()); ++niters) {
244  SOPT_LOW_LOG(" - [PADMM] Iteration {}/{}", niters, itermax());
245  iteration_step(out, residual, lambda, z);
246  SOPT_LOW_LOG(" - [PADMM] Sum of residuals: {}", residual.array().abs().sum());
247  converged = is_converged(out, residual);
248  }
249 
250  if(converged) {
251  SOPT_MEDIUM_LOG(" - [PADMM] converged in {} of {} iterations", niters, itermax());
252  } else if(static_cast<bool>(is_converged())) {
253  // not meaningful if not convergence function
254  SOPT_ERROR(" - [PADMM] did not converge within {} iterations", itermax());
255  }
256  return {niters, converged, std::move(residual)};
257 }
