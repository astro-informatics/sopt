253 template <class SCALAR>
254 typename L2ForwardBackward<SCALAR>::Diagnostic L2ForwardBackward<SCALAR>::operator()(
255  t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
256  SOPT_HIGH_LOG("Performing Forward Backward with L2 and L2 norms");
257  // The f proximal is an L2 proximal
258  Diagnostic result;
259  auto const g_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
260  if (this->l2_proximal_weights().size() > 1)
261  this->l2_proximal_weighted()(out, this->l2_proximal_weights() * gamma, x);
262  else
263  this->l2_proximal()(out, this->l2_proximal_weights()(0) * gamma, x);
264  };
265  const Real sigma_factor = sigma() * sigma();
266  auto const f_gradient = [this, sigma_factor](t_Vector &out, t_Vector const &x) {
267  this->l2_gradient()(out, x / sigma_factor);
268  };
269  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
270  "Objective function");
271  auto const convergence = [this, &scalvar](t_Vector const &x, t_Vector const &residual) mutable {
272  const bool result = this->is_converged(scalvar, x, residual);
273  this->objmin_ = std::real(scalvar.previous());
274  return result;
275  };
276  auto const fb = ForwardBackward<SCALAR>(f_gradient, g_proximal, target())
277  .itermax(itermax())
278  .beta(beta())
279  .gamma(gamma())
280  .nu(nu())
281  .Phi(Phi())
282  .is_converged(convergence);
283  static_cast<typename ForwardBackward<SCALAR>::Diagnostic &>(result) =
284  fb(out, std::tie(guess, res));
285  return result;
286 }

298 template <class SCALAR>
299 bool L2ForwardBackward<SCALAR>::objective_convergence(ScalarRelativeVariation<Scalar> &scalvar,
300  t_Vector const &x,
301  t_Vector const &residual) const {
302  if (static_cast<bool>(objective_convergence())) return objective_convergence()(x, residual);
303  if (scalvar.relative_tolerance() <= 0e0) return true;
304  auto const current = ((gamma() > 0) ? sopt::l2_norm(x, l2_proximal_weights()) * gamma() : 0) +
305  std::pow(sopt::l2_norm(residual), 2) / (2 * sigma() * sigma());
306  return scalvar(current);
307 };
