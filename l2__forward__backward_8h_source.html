254 template <typename SCALAR>
255 typename L2ForwardBackward<SCALAR>::Diagnostic L2ForwardBackward<SCALAR>::operator()(
256  t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
257  SOPT_HIGH_LOG("Performing Forward Backward with L2 and L2 norms");
258  // The f proximal is an L2 proximal
259  Diagnostic result;
260  auto const g_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
261  if (this->l2_proximal_weights().size() > 1)
262  this->l2_proximal_weighted()(out, this->l2_proximal_weights() * gamma, x);
263  else
264  this->l2_proximal()(out, this->l2_proximal_weights()(0) * gamma, x);
265  };
266  const Real sigma_factor = sigma() * sigma();
267  auto const f_gradient = [this, sigma_factor](t_Vector &out, t_Vector const &x) {
268  this->l2_gradient()(out, x / sigma_factor);
269  };
270  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
271  "Objective function");
272  auto const convergence = [this, &scalvar](t_Vector const &x, t_Vector const &residual) mutable {
273  const bool result = this->is_converged(scalvar, x, residual);
274  this->objmin_ = std::real(scalvar.previous());
275  return result;
276  };
277  auto const fb = ForwardBackward<SCALAR>(f_gradient, g_proximal, target())
278  .itermax(itermax())
279  .beta(beta())
280  .gamma(gamma())
281  .nu(nu())
282  .Phi(Phi())
283  .is_converged(convergence);
284  static_cast<typename ForwardBackward<SCALAR>::Diagnostic &>(result) =
285  fb(out, std::tie(guess, res));
286  return result;
287 }

290 bool L2ForwardBackward<SCALAR>::residual_convergence(t_Vector const &x,
291  t_Vector const &residual) const {
292  if (static_cast<bool>(residual_convergence())) return residual_convergence()(x, residual);
293  if (residual_tolerance() <= 0e0) return true;
294  auto const residual_norm = sopt::l2_norm(residual);

299 template <typename SCALAR>
300 bool L2ForwardBackward<SCALAR>::objective_convergence(ScalarRelativeVariation<Scalar> &scalvar,
301  t_Vector const &x,
302  t_Vector const &residual) const {
303  if (static_cast<bool>(objective_convergence())) return objective_convergence()(x, residual);
304  if (scalvar.relative_tolerance() <= 0e0) return true;
305  auto const current = ((gamma() > 0) ? sopt::l2_norm(x, l2_proximal_weights()) * gamma() : 0) +
306  std::pow(sopt::l2_norm(residual), 2) / (2 * sigma() * sigma());
307  return scalvar(current);
308 }
