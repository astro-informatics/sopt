229 template <typename SCALAR>
230 void ProximalADMM<SCALAR>::iteration_step(t_Vector &out, t_Vector &residual, t_Vector &lambda,
231  t_Vector &z) const {
232  g_proximal(z, gamma(), -lambda - residual);
233  f_proximal(out, gamma() / nu(),
234  out - static_cast<t_Vector>(Phi().adjoint() * (residual + lambda + z)) / nu());
235  residual = static_cast<t_Vector>(Phi() * out - target());
236  lambda += lagrange_update_scale() * (residual + z);
237 }

239 template <typename SCALAR>
240 typename ProximalADMM<SCALAR>::Diagnostic ProximalADMM<SCALAR>::operator()(
241  t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
242  SOPT_HIGH_LOG("Performing Proximal ADMM");
243  sanity_check(x_guess, res_guess);
244 
245  t_Vector lambda = t_Vector::Zero(target().size());
246  t_Vector z = t_Vector::Zero(target().size());
247  t_Vector residual = res_guess;
248  out = x_guess;
249 
250  t_uint niters(0);
251  bool converged = false;
252  for (; (not converged) && (niters < itermax()); ++niters) {
253  SOPT_LOW_LOG(" - [PADMM] Iteration {}/{}", niters, itermax());
254  iteration_step(out, residual, lambda, z);
255  SOPT_LOW_LOG(" - [PADMM] Sum of residuals: {}", residual.array().abs().sum());
256  converged = is_converged(out, residual);
257  }
258 
259  if (converged) {
260  SOPT_MEDIUM_LOG(" - [PADMM] converged in {} of {} iterations", niters, itermax());
261  } else if (static_cast<bool>(is_converged())) {
262  // not meaningful if not convergence function
263  SOPT_ERROR(" - [PADMM] did not converge within {} iterations", itermax());
264  }
265  return {niters, converged, std::move(residual)};
