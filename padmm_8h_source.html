228 template <class SCALAR>
229 void ProximalADMM<SCALAR>::iteration_step(t_Vector &out, t_Vector &residual, t_Vector &lambda,
230  t_Vector &z) const {
231  g_proximal(z, gamma(), -lambda - residual);
232  f_proximal(out, gamma() / nu(), out - Phi().adjoint() * (residual + lambda + z) / nu());
233  residual = Phi() * out - target();
234  lambda += lagrange_update_scale() * (residual + z);
235 }

238 typename ProximalADMM<SCALAR>::Diagnostic ProximalADMM<SCALAR>::operator()(
239  t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
240  SOPT_HIGH_LOG("Performing Proximal ADMM");
241  sanity_check(x_guess, res_guess);
242 
243  t_Vector lambda = t_Vector::Zero(target().size());
244  t_Vector z = t_Vector::Zero(target().size());
245  t_Vector residual = res_guess;
246  out = x_guess;
247 
248  t_uint niters(0);
249  bool converged = false;
250  for (; (not converged) && (niters < itermax()); ++niters) {
251  SOPT_LOW_LOG(" - [PADMM] Iteration {}/{}", niters, itermax());
252  iteration_step(out, residual, lambda, z);
253  SOPT_LOW_LOG(" - [PADMM] Sum of residuals: {}", residual.array().abs().sum());
254  converged = is_converged(out, residual);
255  }
256 
257  if (converged) {
258  SOPT_MEDIUM_LOG(" - [PADMM] converged in {} of {} iterations", niters, itermax());
259  } else if (static_cast<bool>(is_converged())) {
260  // not meaningful if not convergence function
261  SOPT_ERROR(" - [PADMM] did not converge within {} iterations", itermax());
262  }
263  return {niters, converged, std::move(residual)};
