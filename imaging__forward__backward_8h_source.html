285 typename ImagingForwardBackward<SCALAR>::Diagnostic ImagingForwardBackward<SCALAR>::operator()(
286  t_Vector &out, t_Vector const &guess, t_Vector const &res) {
287  if(!g_function_)
288  {
289  throw std::runtime_error("Non-differentiable function `g` has not been set. You must set it with `g_function()` before calling the algorithm.");
290  }
291  g_function_->log_message();
292  Diagnostic result;
293  auto const g_proximal = g_function_->proximal_operator();
294  t_Gradient f_gradient;
295  if(f_function_) f_gradient = f_function_->gradient();
296  if(!f_gradient)
297  {
298  SOPT_HIGH_LOG("Gradient function has not been set; using default (gaussian likelihood) gradient. (To set a custom gradient set_gradient() must be called before the algorithm is run.)");
299  f_gradient = [this](t_Vector &output, t_Vector const &x, t_Vector const &residual, t_LinearTransform const &Phi) {
300  output = Phi.adjoint() * (residual / (this->sigma() * this->sigma()));
301  };
302  }
303  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
304  "Objective function");
305  auto const convergence = [this, &scalvar](t_Vector const &x, t_Vector const &residual) mutable {
306  const bool result = this->is_converged(scalvar, x, residual);
307  this->objmin_ = std::real(scalvar.previous());
308  return result;
309  };
310  auto const fb = ForwardBackward<SCALAR>(f_gradient, g_proximal, target())
311  .itermax(itermax())
312  .beta(beta())
313  .gamma(gamma())
314  .nu(nu())
315  .fista(fista())
316  .Phi(Phi())
317  .is_converged(convergence);
318  static_cast<typename ForwardBackward<SCALAR>::Diagnostic &>(result) =
319  fb(out, std::tie(guess, res));
320  return result;
321 }

324 bool ImagingForwardBackward<SCALAR>::residual_convergence(t_Vector const &x,
325  t_Vector const &residual) const {
326  if (static_cast<bool>(residual_convergence())) return residual_convergence()(x, residual);
327  if (residual_tolerance() <= 0e0) return true;
328  auto const residual_norm = sopt::l2_norm(residual);
329  SOPT_LOW_LOG(" - [FB] Residuals: {} <? {}", residual_norm, residual_tolerance());
330  return residual_norm < residual_tolerance();

333 template <typename SCALAR>
334 bool ImagingForwardBackward<SCALAR>::objective_convergence(ScalarRelativeVariation<Scalar> &scalvar,
335  t_Vector const &x,
336  t_Vector const &residual) const {
337  if (static_cast<bool>(objective_convergence())) return objective_convergence()(x, residual);
338  if (scalvar.relative_tolerance() <= 0e0) return true;
339  auto const current = ((gamma() > 0) ? g_function_->function(x)
340  * gamma() : 0) + std::pow(sopt::l2_norm(residual), 2) / (2 * sigma() * sigma());
341  return scalvar(current);
342 }

359 template <typename SCALAR>
360 bool ImagingForwardBackward<SCALAR>::is_converged(ScalarRelativeVariation<Scalar> &scalvar,
361  t_Vector const &x,
362  t_Vector const &residual) const {
363  auto const user = static_cast<bool>(is_converged()) == false or is_converged()(x, residual);
364  auto const res = residual_convergence(x, residual);
365 #ifdef SOPT_MPI
366  auto const obj = objective_convergence(obj_comm(), scalvar, x, residual);
367 #else
368  auto const obj = objective_convergence(scalvar, x, residual);
369 #endif
370  // beware of short-circuiting!
371  // better evaluate each convergence function everytime, especially with mpi
372  return user and res and obj;
