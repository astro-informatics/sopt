309 template <class SCALAR>
310 typename ImagingForwardBackward<SCALAR>::Diagnostic ImagingForwardBackward<SCALAR>::operator()(
311  t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
312  SOPT_HIGH_LOG("Performing Forward Backward with L1 and L2 norms");
313  // The f proximal is an L1 proximal that stores some diagnostic result
314  Diagnostic result;
315  auto const g_proximal = [this, &result](t_Vector &out, Real gamma, t_Vector const &x) {
316  result.l1_diagnostic = this->l1_proximal(out, gamma, x);
317  };
318  const Real sigma_factor = sigma() * sigma();
319  auto const f_gradient = [this, sigma_factor](t_Vector &out, t_Vector const &x) {
320  this->l2_gradient()(out, x / sigma_factor);
321  };
322  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
323  "Objective function");
324  auto const convergence = [this, &scalvar](t_Vector const &x, t_Vector const &residual) mutable {
325  const bool result = this->is_converged(scalvar, x, residual);
326  this->objmin_ = std::real(scalvar.previous());
327  return result;
328  };
329  auto const fb = ForwardBackward<SCALAR>(f_gradient, g_proximal, target())
330  .itermax(itermax())
331  .beta(beta())
332  .gamma(gamma())
333  .nu(nu())
334  .Phi(Phi())
335  .is_converged(convergence);
336  static_cast<typename ForwardBackward<SCALAR>::Diagnostic &>(result) =
337  fb(out, std::tie(guess, res));
338  return result;
339 }

343  t_Vector const &residual) const {

351 template <class SCALAR>
352 bool ImagingForwardBackward<SCALAR>::objective_convergence(ScalarRelativeVariation<Scalar> &scalvar,
353  t_Vector const &x,
354  t_Vector const &residual) const {
355  if (static_cast<bool>(objective_convergence())) return objective_convergence()(x, residual);
356  if (scalvar.relative_tolerance() <= 0e0) return true;
357  auto const current = ((gamma() > 0) ? sopt::l1_norm(static_cast<t_Vector>(Psi().adjoint() * x),
358  l1_proximal_weights()) *
359  gamma()
360  : 0) +
361  std::pow(sopt::l2_norm(residual), 2) / (2 * sigma() * sigma());
362  return scalvar(current);
363 };

383 template <class SCALAR>
384 bool ImagingForwardBackward<SCALAR>::is_converged(ScalarRelativeVariation<Scalar> &scalvar,
385  t_Vector const &x,
386  t_Vector const &residual) const {
387  auto const user = static_cast<bool>(is_converged()) == false or is_converged()(x, residual);
388  auto const res = residual_convergence(x, residual);
389 #ifdef SOPT_MPI
390  auto const obj = objective_convergence(obj_comm(), scalvar, x, residual);
391 #else
392  auto const obj = objective_convergence(scalvar, x, residual);
393 #endif
394  // beware of short-circuiting!
395  // better evaluate each convergence function everytime, especially with mpi
396  return user and res and obj;
