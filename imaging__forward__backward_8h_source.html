260 typename ImagingForwardBackward<SCALAR>::Diagnostic ImagingForwardBackward<SCALAR>::operator()(
261  t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
262  g_proximal_->log_message();
263  // The f proximal is an L1 proximal that stores some diagnostic result
264  Diagnostic result;
265  auto const g_proximal_function = g_proximal_->proximal_function();
266  const Real sigma_factor = sigma() * sigma();
267  auto const f_gradient = [this, sigma_factor](t_Vector &out, t_Vector const &x) {
268  this->l2_gradient()(out, x / sigma_factor);
269  };
270  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
271  "Objective function");
272  auto const convergence = [this, &scalvar](t_Vector const &x, t_Vector const &residual) mutable {
273  const bool result = this->is_converged(scalvar, x, residual);
274  this->objmin_ = std::real(scalvar.previous());
275  return result;
276  };
277  auto const fb = ForwardBackward<SCALAR>(f_gradient, g_proximal_function, target())
278  .itermax(itermax())
279  .beta(beta())
280  .gamma(gamma())
281  .nu(nu())
282  .fista(fista())
283  .Phi(Phi())
284  .is_converged(convergence);
285  static_cast<typename ForwardBackward<SCALAR>::Diagnostic &>(result) =
286  fb(out, std::tie(guess, res));

291 bool ImagingForwardBackward<SCALAR>::residual_convergence(t_Vector const &x,
292  t_Vector const &residual) const {
293  if (static_cast<bool>(residual_convergence())) return residual_convergence()(x, residual);
294  if (residual_tolerance() <= 0e0) return true;
295  auto const residual_norm = sopt::l2_norm(residual);
296  SOPT_LOW_LOG(" - [FB] Residuals: {} <? {}", residual_norm, residual_tolerance());
297  return residual_norm < residual_tolerance();

300 template <typename SCALAR>
301 bool ImagingForwardBackward<SCALAR>::objective_convergence(ScalarRelativeVariation<Scalar> &scalvar,
302  t_Vector const &x,
303  t_Vector const &residual) const {
304  if (static_cast<bool>(objective_convergence())) return objective_convergence()(x, residual);
305  if (scalvar.relative_tolerance() <= 0e0) return true;
306  auto const current = ((gamma() > 0) ? g_proximal_->proximal_norm(x)
307  * gamma() : 0) + std::pow(sopt::l2_norm(residual), 2) / (2 * sigma() * sigma());
308  return scalvar(current);
309 }
