201 template <class ALGORITHM>
202 typename Reweighted<ALGORITHM>::ReweightedResult Reweighted<ALGORITHM>::operator()(
203  ReweightedResult const &warm) const {
204  SOPT_HIGH_LOG("Starting reweighted scheme");
205  // Copies inner algorithm, so that operator() can be constant
206  Algorithm algo(algorithm());
207  ReweightedResult result(warm);
208 
209  auto delta = std::max(standard_deviation(reweightee(warm.algo.x)), min_delta());
210  SOPT_LOW_LOG("- Initial delta: {}", delta);
211  for (result.niters = 0; result.niters < itermax(); ++result.niters) {
212  SOPT_LOW_LOG("Reweigting iteration {}/{} ", result.niters, itermax());
213  SOPT_LOW_LOG(" - delta: {}", delta);
214  result.weights = delta / (delta + reweightee(result.algo.x).array().abs());
215  set_weights(algo, result.weights);
216  result.algo = algo(result.algo);
217  if (is_converged(result.algo.x)) {
218  SOPT_MEDIUM_LOG("Reweighting scheme did converge in {} iterations", result.niters);
219  result.good = true;
220  break;
221  }
222  delta = std::max(min_delta(), update_delta(delta));
223  }
224  // result is always good if no convergence function is defined
225  if (not is_converged())
226  result.good = true;
227  else if (not result.good)
228  SOPT_ERROR("Reweighting scheme did *not* converge in {} iterations", itermax());
229  return result;
230 }
