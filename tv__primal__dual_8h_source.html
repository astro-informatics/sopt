301 template <typename SCALAR>
302 typename TVPrimalDual<SCALAR>::Diagnostic TVPrimalDual<SCALAR>::operator()(
303  t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
304  SOPT_HIGH_LOG("Performing Primal Dual with TV and L2 operators");
305  // The f proximal is an L1 proximal that stores some diagnostic result
306  if (not check_tv_weight_proximal(tv_proximal(), tv_proximal_weighted()))
307  SOPT_THROW(
308  "tv proximal and weighted tv proximal appear to be different functions. Please make sure "
309  "both are the same function.");
310  auto const f_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
311  if (this->tv_proximal_weights().size() > 1)
312  this->tv_proximal_weighted()(out, this->tv_proximal_weights() * gamma, x);
313  else
314  this->tv_proximal()(out, this->tv_proximal_weights()(0) * gamma, x);
315  };
316  auto const g_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
317  this->l2ball_proximal()(out, gamma, x);
318  // applying preconditioning
319  for (t_int i = 0; i < this->precondition_iters(); i++)
320  this->l2ball_proximal()(
321  out, gamma,
322  out - this->precondition_stepsize() *
323  (out.array() * this->precondition_weights().array() - x.array()).matrix());
324 
325  if (this->precondition_iters() > 0) out = out.array() * this->precondition_weights().array();
326  };
327  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
328  "Objective function");
329  auto const convergence = [this, scalvar](t_Vector const &x, t_Vector const &residual) mutable {
330  return this->is_converged(scalvar, x, residual);
331  };
332  const bool positive = positivity_constraint();
333  const bool real = real_constraint();
334  t_Constraint constraint = [real, positive](t_Vector &out, const t_Vector &x) {
335  if (real) out.real() = x.real();
336  if (positive) out = sopt::positive_quadrant(x);
337  if (not real and not positive) out = x;
338  };
339  auto const pd = PD(f_proximal, g_proximal, target())
340  .itermax(itermax())
341  .constraint(constraint)
342  .sigma(sigma())
343  .tau(tau())
344  .gamma(gamma())
345  .update_scale(update_scale())
346  .xi(xi())
347  .rho(rho())
348  .nu(nu())
349  .gamma(gamma())
350  .Phi(Phi())
351  .Psi(Psi())
352  .random_measurement_updater(random_measurement_updater())
353  .random_wavelet_updater(random_wavelet_updater())
354 #ifdef SOPT_MPI
355  .v_all_sum_all_comm(v_all_sum_all_comm())
356  .u_all_sum_all_comm(u_all_sum_all_comm())
357 #endif
358  .is_converged(convergence);
359  Diagnostic result;
360  static_cast<typename PD::Diagnostic &>(result) = pd(out, std::tie(guess, res));
361  return result;
362 }

364 template <typename SCALAR>
365 bool TVPrimalDual<SCALAR>::residual_convergence(t_Vector const &x, t_Vector const &residual) const {
366  if (static_cast<bool>(residual_convergence())) return residual_convergence()(x, residual);
367  if (residual_tolerance() <= 0e0) return true;
368  auto const residual_norm = sopt::l2_norm(residual, l2ball_proximal_weights());
369  SOPT_LOW_LOG(" - [Primal Dual] Residuals: {} <? {}", residual_norm, residual_tolerance());
370  return residual_norm < residual_tolerance();

373 template <typename SCALAR>
374 bool TVPrimalDual<SCALAR>::objective_convergence(ScalarRelativeVariation<Scalar> &scalvar,
375  t_Vector const &x,
376  t_Vector const &residual) const {

384 template <typename SCALAR>
385 bool TVPrimalDual<SCALAR>::is_converged(ScalarRelativeVariation<Scalar> &scalvar, t_Vector const &x,
386  t_Vector const &residual) const {
387  auto const user = static_cast<bool>(is_converged()) == false or is_converged()(x, residual);
388  auto const res = residual_convergence(x, residual);
389  auto const obj = objective_convergence(scalvar, x, residual);
390  // beware of short-circuiting!
391  // better evaluate each convergence function everytime, especially with mpi
392  return user and res and obj;
393 }
