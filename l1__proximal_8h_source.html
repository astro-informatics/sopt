332 template <class SCALAR>
333 template <class T0, class MIXING>
334 typename L1<SCALAR>::Diagnostic L1<SCALAR>::operator()(Eigen::MatrixBase<T0> &out, Real gamma,
335  Vector<Scalar> const &x,
336  MIXING mixing) const {
337  SOPT_MEDIUM_LOG("Starting Proximal L1 operator:");
338  t_uint niters = 0;
339  out = x;
340 
341  Breaker breaker(objective(out, x, gamma), tolerance(), false); // not fista_mixing());
342  SOPT_LOW_LOG(" - [ProxL1] iter {}, prox_fval = {}", niters, breaker.current());
343  Vector<Scalar> const res = Psi().adjoint() * out;
344  Vector<Scalar> u_l1 = 1e0 / nu() * (res - apply_soft_threshhold(gamma, res));
345  apply_constraints(out, x - Psi() * u_l1);
346 
347  // Move on to other iterations
348  for (++niters; niters < itermax() or itermax() == 0; ++niters) {
349  auto const do_break = breaker(objective(x, out, gamma));
350  SOPT_LOW_LOG(" - [ProxL1] iter {}, prox_fval = {}, rel_fval = {}", niters, breaker.current(),
351  breaker.relative_variation());
352  if (do_break) break;
353 
354  Vector<Scalar> const res = u_l1 * nu() + Psi().adjoint() * out;
355  mixing(u_l1, 1e0 / nu() * (res - apply_soft_threshhold(gamma, res)), niters);
356  apply_constraints(out, x - Psi() * u_l1);
357  }
358 
359  if (breaker.two_cycle()) SOPT_WARN("Two-cycle detected when computing L1");
360 
361  if (breaker.converged()) {
362  SOPT_LOW_LOG("Proximal L1 operator converged at {} in {} iterations", breaker.current(),
363  niters);
364  } else
365  SOPT_ERROR("Proximal L1 operator did not converge after {} iterations", niters);
366  return {niters, breaker.relative_variation(), breaker.current(), breaker.converged()};
367 }

395  template <class T1>
396  void operator()(Vector<SCALAR> &previous, Eigen::MatrixBase<T1> const &unmixed, t_uint iter) {
397  // reset
398  if (iter == 0) {
399  previous = unmixed;
400  return;
401  }
402  if (iter <= 1) t = next(1);
403  auto const prior_t = t;
404  t = next(t);
405  auto const alpha = (prior_t - 1) / t;
406  previous = (1e0 + alpha) * unmixed.derived() - alpha * previous;
407  }
408  static Real next(Real t) { return 0.5 + 0.5 * std::sqrt(1e0 + 4e0 * t * t); }
