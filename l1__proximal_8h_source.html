327 template <class SCALAR>
328 template <class T0, class MIXING>
329 typename L1<SCALAR>::Diagnostic L1<SCALAR>::operator()(Eigen::MatrixBase<T0> &out, Real gamma,
330  Vector<Scalar> const &x,
331  MIXING mixing) const {
332  SOPT_MEDIUM_LOG("Starting Proximal L1 operator:");
333  t_uint niters = 0;
334  out = x;
335 
336  Breaker breaker(objective(out, x, gamma), tolerance(), false); // not fista_mixing());
337  SOPT_LOW_LOG(" - [ProxL1] iter {}, prox_fval = {}", niters, breaker.current());
338  Vector<Scalar> const res = Psi().adjoint() * out;
339  Vector<Scalar> u_l1 = 1e0 / nu() * (res - apply_soft_threshhold(gamma, res));
340  apply_constraints(out, x - Psi() * u_l1);
341 
342  // Move on to other iterations
343  for (++niters; niters < itermax() or itermax() == 0; ++niters) {
344  auto const do_break = breaker(objective(x, out, gamma));
345  SOPT_LOW_LOG(" - [ProxL1] iter {}, prox_fval = {}, rel_fval = {}", niters, breaker.current(),
346  breaker.relative_variation());
347  if (do_break) break;
348 
349  Vector<Scalar> const res = u_l1 * nu() + Psi().adjoint() * out;
350  mixing(u_l1, 1e0 / nu() * (res - apply_soft_threshhold(gamma, res)), niters);
351  apply_constraints(out, x - Psi() * u_l1);
352  }
353 
354  if (breaker.two_cycle()) SOPT_WARN("Two-cycle detected when computing L1");
355 
356  if (breaker.converged()) {
357  SOPT_LOW_LOG("Proximal L1 operator converged at {} in {} iterations", breaker.current(),
358  niters);
359  } else
360  SOPT_ERROR("Proximal L1 operator did not converge after {} iterations", niters);
361  return {niters, breaker.relative_variation(), breaker.current(), breaker.converged()};
362 }

390  template <class T1>
391  void operator()(Vector<SCALAR> &previous, Eigen::MatrixBase<T1> const &unmixed, t_uint iter) {
392  // reset
393  if (iter == 0) {
394  previous = unmixed;
395  return;
396  }
397  if (iter <= 1) t = next(1);
398  auto const prior_t = t;
399  t = next(t);
400  auto const alpha = (prior_t - 1) / t;
401  previous = (1e0 + alpha) * unmixed.derived() - alpha * previous;
402  }
403  static Real next(Real t) { return 0.5 + 0.5 * std::sqrt(1e0 + 4e0 * t * t); }
