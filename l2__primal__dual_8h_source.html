302 template <class SCALAR>
303 typename ImagingPrimalDual<SCALAR>::Diagnostic ImagingPrimalDual<SCALAR>::operator()(
304  t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
305  SOPT_HIGH_LOG("Performing Primal Dual with L1 and L2 operators");
306  // The f proximal is an L1 proximal that stores some diagnostic result
307  if (not check_l2_weight_proximal(l2_proximal(), l2_proximal_weighted()))
308  SOPT_THROW(
309  "l1 proximal and weighted l1 proximal appear to be different functions. Please make sure "
310  "both are the same function.");
311  auto const f_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
312  if (this->l2_proximal_weights().size() > 1)
313  this->l2_proximal_weighted()(out, this->l2_proximal_weights() * gamma, x);
314  else
315  this->l2_proximal()(out, this->l2_proximal_weights()(0) * gamma, x);
316  };
317  auto const g_proximal = [this](t_Vector &out, Real gamma, t_Vector const &x) {
318  this->l2ball_proximal()(out, gamma, x);
319  // applying preconditioning
320  for (t_int i = 0; i < this->precondition_iters(); i++)
321  this->l2ball_proximal()(
322  out, gamma,
323  out - this->precondition_stepsize() *
324  (out.array() * this->precondition_weights().array() - x.array()).matrix());
325 
326  if (this->precondition_iters() > 0) out = out.array() * this->precondition_weights().array();
327  };
328  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
329  "Objective function");
330  auto const convergence = [this, scalvar](t_Vector const &x, t_Vector const &residual) mutable {
331  return this->is_converged(scalvar, x, residual);
332  };
333  const bool positive = positivity_constraint();
334  const bool real = real_constraint();
335  t_Constraint constraint = [real, positive](t_Vector &out, const t_Vector &x) {
336  if (real) out.real() = x.real();
337  if (positive) out = sopt::positive_quadrant(x);
338  if (not real and not positive) out = x;
339  };
340  auto const pd = PD(f_proximal, g_proximal, target())
341  .itermax(itermax())
342  .constraint(constraint)
343  .sigma(sigma())
344  .tau(tau())
345  .gamma(gamma())
346  .update_scale(update_scale())
347  .xi(xi())
348  .rho(rho())
349  .nu(nu())
350  .gamma(gamma())
351  .Phi(Phi())
352  .Psi(Psi())
353  .random_measurement_updater(random_measurement_updater())
354  .random_wavelet_updater(random_wavelet_updater())
355 #ifdef SOPT_MPI
356  .v_all_sum_all_comm(v_all_sum_all_comm())
357  .u_all_sum_all_comm(u_all_sum_all_comm())
358 #endif
359  .is_converged(convergence);
360  Diagnostic result;
361  static_cast<typename PD::Diagnostic &>(result) = pd(out, std::tie(guess, res));
362  return result;
363 }

366 bool ImagingPrimalDual<SCALAR>::residual_convergence(t_Vector const &x,
367  t_Vector const &residual) const {
368  if (static_cast<bool>(residual_convergence())) return residual_convergence()(x, residual);
369  if (residual_tolerance() <= 0e0) return true;
370  auto const residual_norm = sopt::l2_norm(residual, l2ball_proximal_weights());
371  SOPT_LOW_LOG(" - [Primal Dual] Residuals: {} <? {}", residual_norm, residual_tolerance());
372  return residual_norm < residual_tolerance();

375 template <class SCALAR>
376 bool ImagingPrimalDual<SCALAR>::objective_convergence(ScalarRelativeVariation<Scalar> &scalvar,
377  t_Vector const &x,
378  t_Vector const &residual) const {
379  if (static_cast<bool>(objective_convergence())) return objective_convergence()(x, residual);
380  if (scalvar.relative_tolerance() <= 0e0) return true;
381  auto const current =
382  (l2_proximal_weights().size() > 1)
383  ? sopt::l2_norm(l2_proximal_weights().array() * (Psi().adjoint() * x).array())
384  : sopt::l2_norm(l2_proximal_weights()(0) * (Psi().adjoint() * x));
385  return scalvar(current);
386 };

388 template <class SCALAR>
389 bool ImagingPrimalDual<SCALAR>::is_converged(ScalarRelativeVariation<Scalar> &scalvar,
390  t_Vector const &x, t_Vector const &residual) const {
391  auto const user = static_cast<bool>(is_converged()) == false or is_converged()(x, residual);
392  auto const res = residual_convergence(x, residual);
393  auto const obj = objective_convergence(scalvar, x, residual);
394  // beware of short-circuiting!
395  // better evaluate each convergence function everytime, especially with mpi
396  return user and res and obj;
397 }
