255 void ForwardBackward<SCALAR>::iteration_step(t_Vector &image, t_Vector &residual, t_Vector &auxilliary_image,
256  t_Vector &gradient_current, const t_real FISTA_step) const {
257  t_Vector prev_image = image;
258  f_gradient(gradient_current, auxilliary_image, residual, Phi()); // takes residual and calculates the grad = 1/sig^2 residual
259  t_Vector auxilliary_with_step = auxilliary_image - beta() / nu() * gradient_current; // step to new image using gradient
260  const Real weight = gamma() * beta();
261  g_proximal(image, weight, auxilliary_with_step); // apply proximal operator to new image
262  auxilliary_image = image + FISTA_step * (image - prev_image); // update auxilliary vector with FISTA acceleration step
263  residual = (Phi() * auxilliary_image) - target(); // updates the residual for the NEXT iteration (new image).
264 }

270  if (fista()) {

272  } else {
273  SOPT_HIGH_LOG("Using standard FB algorithm");
274  }
275  sanity_check(x_guess, res_guess);
276 
277  const size_t image_size = x_guess.size();
278 
279  t_Vector auxilliary_image = x_guess;
280  t_Vector residual = res_guess;
281  t_Vector gradient_current = t_Vector::Zero(image_size);
282  out = x_guess;
283 
284  t_uint niters(0);
285  bool converged = false;
286  Real theta = 1.0;
287  Real theta_new = 1.0;
288  Real FISTA_step = 0.0;
289  for (; (not converged) && (niters < itermax()); ++niters) {
290  SOPT_LOW_LOG(" - [FB] Iteration {}/{}", niters, itermax());
291  if (fista()) {
292  theta_new = (1 + std::sqrt(1 + 4 * theta * theta)) / 2.;
293  FISTA_step = (theta - 1) / (theta_new);
294  theta = theta_new;
295  }
296  SOPT_LOW_LOG(" - Call iteration step");
297  iteration_step(out, residual, auxilliary_image, gradient_current, FISTA_step);
298  SOPT_LOW_LOG(" - [FB] Sum of residuals: {}", residual.array().abs().sum());
299  converged = is_converged(out, residual);
300  }
301 
302  if (converged) {
303  SOPT_MEDIUM_LOG(" - [FB] converged in {} of {} iterations", niters, itermax());
304  } else if (static_cast<bool>(is_converged())) {
305  // not meaningful if not convergence function
306  SOPT_ERROR(" - [FB] did not converge within {} iterations", itermax());
307  }
308  return {niters, converged, std::move(residual)};
309 }
