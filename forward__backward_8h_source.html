233 template <typename SCALAR>
234 void ForwardBackward<SCALAR>::iteration_step(t_Vector &out, t_Vector &residual, t_Vector &p,
235  t_Vector &z, const t_real lambda) const {
236  p = out;
237  f_gradient(z, residual);
238  const t_Vector input = out - beta() / nu() * (Phi().adjoint() * z);
239  const Real weight = gamma() * beta();
240  g_proximal(out, weight, input);
241  p = out + lambda * (out - p);
242  residual = (Phi() * p) - target();
243 }

245 template <typename SCALAR>
246 typename ForwardBackward<SCALAR>::Diagnostic ForwardBackward<SCALAR>::operator()(
247  t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
248  SOPT_HIGH_LOG("Performing Forward Backward Splitting");
249  if (fista()) {
250  SOPT_HIGH_LOG("Using FISTA algorithm");
251  } else {
252  SOPT_HIGH_LOG("Using standard FB algorithm");
253  }
254  sanity_check(x_guess, res_guess);
255 
256  t_Vector p = t_Vector::Zero(x_guess.size());
257  t_Vector z = t_Vector::Zero(target().size());
258  t_Vector residual = res_guess;
259  out = x_guess;
260 
261  t_uint niters(0);
262  bool converged = false;
263  Real theta = 1.0;
264  Real theta_new = 1.0;
265  Real lambda = 0.0;
266  for (; (not converged) && (niters < itermax()); ++niters) {
267  SOPT_LOW_LOG(" - [FB] Iteration {}/{}", niters, itermax());
268  if (fista()) {
269  theta_new = (1 + std::sqrt(1 + 4 * theta * theta)) / 2.;
270  lambda = (theta - 1) / (theta_new);
271  theta = theta_new;
272  }
273  iteration_step(out, residual, p, z, lambda);
274  SOPT_LOW_LOG(" - [FB] Sum of residuals: {}", residual.array().abs().sum());
275  converged = is_converged(out, residual);
276  }
277 
278  if (converged) {
279  SOPT_MEDIUM_LOG(" - [FB] converged in {} of {} iterations", niters, itermax());
280  } else if (static_cast<bool>(is_converged())) {
281  // not meaningful if not convergence function
282  SOPT_ERROR(" - [FB] did not converge within {} iterations", itermax());
283  }
284  return {niters, converged, std::move(residual)};
285 }
