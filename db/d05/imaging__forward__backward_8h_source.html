266 template <class SCALAR>
267 typename ImagingForwardBackward<SCALAR>::Diagnostic ImagingForwardBackward<SCALAR>::
268 operator()(t_Vector &out, t_Vector const &guess, t_Vector const &res) const {
269  SOPT_HIGH_LOG("Performing Foward Backward with L1 operator");
270  // The g proximal is an L1 proximal that stores some diagnostic result
271  Diagnostic result;
272  auto const g_proximal = [this, &result](t_Vector &out, Real gamma, t_Vector const &x) {
273  result.l1_diagnostic = this->l1_proximal(out, gamma, x);
274  };
275  ScalarRelativeVariation<Scalar> scalvar(relative_variation(), relative_variation(),
276  "Objective function");
277  auto const convergence = [this, scalvar](t_Vector const &x, t_Vector const &residual) mutable {
278  return this->is_converged(scalvar, x, residual);
279  };
280  auto const fb = ForwardBackward<SCALAR>(g_proximal, target())
281  .itermax(itermax())
282  .beta(beta())
283  .mu(mu())
284  .sigma(sigma())
285  .PhiTPhi(PhiTPhi())
286  .is_converged(convergence);
287  static_cast<typename FB::Diagnostic &>(result) = fb(out, std::tie(guess, res));
288  return result;
289 }

292 bool ImagingForwardBackward<SCALAR>::residual_convergence(t_Vector const &x,
293  t_Vector const &residual) const {
294  if(static_cast<bool>(residual_convergence()))
295  return residual_convergence()(x, residual);
296  if(residual_tolerance() <= 0e0)
297  return true;
298  auto const residual_norm = sopt::l2_norm(residual);
299  SOPT_LOW_LOG(" - [FB] Residuals: {} <? {}", residual_norm, residual_tolerance());
300  return residual_norm < residual_tolerance();

304 bool ImagingForwardBackward<SCALAR>::objective_convergence(ScalarRelativeVariation<Scalar> &scalvar,
305  t_Vector const &x,
306  t_Vector const &residual) const {
307  if(static_cast<bool>(objective_convergence()))

315 template <class SCALAR>
316 bool ImagingForwardBackward<SCALAR>::is_converged(ScalarRelativeVariation<Scalar> &scalvar,
317  t_Vector const &x,
318  t_Vector const &residual) const {
319  auto const user = static_cast<bool>(is_converged()) == false or is_converged()(x, residual);
320  auto const res = residual_convergence(x, residual);
321  auto const obj = objective_convergence(scalvar, x, residual);
322  // beware of short-circuiting!
323  // better evaluate each convergence function everytime, especially with mpi
324  return user and res and obj;
325 }
