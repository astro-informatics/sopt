224 template <class SCALAR>
225 void PrimalDual<SCALAR>::iteration_step(t_Vector &out, t_Vector &residual, t_Vector &s, t_Vector &v,
226  t_Vector &x_bar) const {
227  t_Vector prev_sol = out;
228  t_Vector prev_s = s;
229  t_Vector prev_v = v;
230 
231  auto const l2ball_proximal = proximal::L2Ball<Scalar>(l2ball_epsilon());
232 
233  // v_t = v_t-1 + Phi*x_bar - l2ball_prox(v_t-1 + Phi*x_bar)
234  t_Vector temp = v + (Phi() * x_bar);
235  t_Vector v_prox;
236  v_prox = l2ball_proximal(0, temp - target()) + target();
237  v = temp - v_prox;
238 
239  // s_t = s_t-1 + Psi_dagger * x_bar_t-1 - l1norm_prox(s_t-1 + Psi_dagger * x_bar_t-1)
240  t_Vector temp2 = s + (Psi().adjoint() * x_bar);
241  t_Vector s_prox;
242  proximal::l1_norm(s_prox, kappa() / sigma1(), temp2);
243  s = temp2 - s_prox;
244 
245  // x_t = positive orth projection(x_t-1 - tau * (sigma1 * Psi * s + sigma2 * Phi dagger * v))
246  out = prev_sol - tau() * (Psi() * s * sigma1() + Phi().adjoint() * v * sigma2());
247  if (positivity_constraint()) {
248  out = sopt::positive_quadrant(out);
249  }
250  x_bar = 2 * out - prev_sol;
251  residual = Phi() * out - target();
252 }

254 template <class SCALAR>
255 typename PrimalDual<SCALAR>::Diagnostic PrimalDual<SCALAR>::operator()(
256  t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
257  SOPT_HIGH_LOG("Performing Primal Dual");
258  sanity_check(x_guess, res_guess);
259 
260  // This should be number of dicitionaries times x_guess.size()
261  // Should be looking up the multiplication factor from Psi
262  t_Vector s = t_Vector::Zero(x_guess.size() * levels());
263  t_Vector v = t_Vector::Zero(target().size());
264 
265  t_Vector x_bar = t_Vector::Zero(x_guess.size());
266  t_Vector residual = res_guess;
267 
268  // Check if there is a user provided convergence function
269  bool const has_user_convergence = static_cast<bool>(is_converged());
270  bool converged = false;
271 
272  out = x_guess;
273  t_uint niters(0);
274 
275  Vector<Real> l1_weights;
276 
277  if (l1_proximal_weights().size() == 1 && (l1_proximal_weights()(0)) == 0) {
278  l1_weights = Vector<Real>::Ones(1);
279 
280  } else {
281  l1_weights = l1_proximal_weights();
282  }
283 
284  Vector<Real> weights = Vector<Real>::Ones(1);
285 
286  std::pair<Real, Real> objectives{sopt::l1_norm(Psi().adjoint() * out, l1_weights), 0};
287 
288  for (; niters < itermax(); ++niters) {
289  SOPT_LOW_LOG(" - [PD] Iteration {}/{}", niters, itermax());
290  iteration_step(out, residual, s, v, x_bar);
291  SOPT_LOW_LOG(" - [PD] Sum of residuals: {}", residual.array().abs().sum());
292 
293  objectives.second = objectives.first;
294  objectives.first = sopt::l1_norm(Psi().adjoint() * out, l1_weights);
295  t_real const relative_objective =
296  std::abs(objectives.first - objectives.second) / objectives.first;
297  SOPT_LOW_LOG(" - [PD] objective: obj value = {}, rel obj = {}", objectives.first,
298  relative_objective);
299 
300  auto const residual_norm = sopt::l2_norm(residual, weights);
301  SOPT_LOW_LOG(" - [PD] residual norm = {}, residual convergence = {}", residual_norm,
302  residual_convergence());
303 
304  auto const user = (not has_user_convergence) or is_converged(out);
305  auto const res = residual_convergence() <= 0e0 or residual_norm < residual_convergence();
306  auto const rel = relative_variation() <= 0e0 or relative_objective < relative_variation();
307 
308  converged = user and res and rel;
309  if (converged) {
310  SOPT_MEDIUM_LOG(" - [PD] converged in {} of {} iterations", niters, itermax());
311  break;
312  }
313  }
314  // check function exists, otherwise, don't know if convergence is meaningful
315  if (not converged) SOPT_ERROR(" - [PD] did not converge within {} iterations", itermax());
316 
317  return {niters, converged, std::move(residual)};
318 }
