211 template <class SCALAR>
212 void PrimalDual<SCALAR>::iteration_step(t_Vector &out, t_Vector &residual, t_Vector &s, t_Vector &v,
213  t_Vector &x_bar) const {
214 
215  t_Vector prev_sol = out;
216  t_Vector prev_s = s;
217  t_Vector prev_v = v;
218 
219  auto const l2ball_proximal = proximal::L2Ball<Scalar>(l2ball_epsilon());
220 
221  // v_t = v_t-1 + Phi*x_bar - l2ball_prox(v_t-1 + Phi*x_bar)
222  t_Vector temp = v + (Phi() * x_bar);
223  t_Vector v_prox;
224  v_prox = l2ball_proximal(0, temp - target()) + target();
225  v = temp - v_prox;
226 
227  // s_t = s_t-1 + Psi_dagger * x_bar_t-1 - l1norm_prox(s_t-1 + Psi_dagger * x_bar_t-1)
228  t_Vector temp2 = s + (Psi().adjoint() * x_bar);
229  t_Vector s_prox;
230  proximal::l1_norm(s_prox, kappa() / sigma1(), temp2);
231  s = temp2 - s_prox;
232 
233  // x_t = positive orth projection(x_t-1 - tau * (sigma1 * Psi * s + sigma2 * Phi dagger * v))
234  out = prev_sol - tau() * (Psi() * s * sigma1() + Phi().adjoint() * v * sigma2());
235  if(positivity_constraint()) {
236  out = sopt::positive_quadrant(out);
237  }
238  x_bar = 2 * out - prev_sol;
239  residual = Phi() * out - target();
240 }

242 template <class SCALAR>
243 typename PrimalDual<SCALAR>::Diagnostic PrimalDual<SCALAR>::
244 operator()(t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
245  SOPT_HIGH_LOG("Performing Primal Dual");
246  sanity_check(x_guess, res_guess);
247 
248  // This should be number of dicitionaries times x_guess.size()
249  // Should be looking up the multiplication factor from Psi
250  t_Vector s = t_Vector::Zero(x_guess.size() * levels());
251  t_Vector v = t_Vector::Zero(target().size());
252 
253  t_Vector x_bar = t_Vector::Zero(x_guess.size());
254  t_Vector residual = res_guess;
255 
256  // Check if there is a user provided convergence function
257  bool const has_user_convergence = static_cast<bool>(is_converged());
258  bool converged = false;
259 
260  out = x_guess;
261  t_uint niters(0);
262 
263  Vector<Real> l1_weights;
264 
265  if(l1_proximal_weights().size() == 1 && (l1_proximal_weights()(0)) == 0) {
266  l1_weights = Vector<Real>::Ones(1);
267 
268  } else {
269  l1_weights = l1_proximal_weights();
270  }
271 
272  Vector<Real> weights = Vector<Real>::Ones(1);
273 
274  std::pair<Real, Real> objectives{sopt::l1_norm(Psi().adjoint() * out, l1_weights), 0};
275 
276  for(; niters < itermax(); ++niters) {
277  SOPT_LOW_LOG(" - [PD] Iteration {}/{}", niters, itermax());
278  iteration_step(out, residual, s, v, x_bar);
279  SOPT_LOW_LOG(" - [PD] Sum of residuals: {}", residual.array().abs().sum());
280 
281  objectives.second = objectives.first;
282  objectives.first = sopt::l1_norm(Psi().adjoint() * out, l1_weights);
283  t_real const relative_objective
284  = std::abs(objectives.first - objectives.second) / objectives.first;
285  SOPT_LOW_LOG(" - [PD] objective: obj value = {}, rel obj = {}", objectives.first,
286  relative_objective);
287 
288  auto const residual_norm = sopt::l2_norm(residual, weights);
289  SOPT_LOW_LOG(" - [PD] residual norm = {}, residual convergence = {}", residual_norm,
290  residual_convergence());
291 
292  auto const user = (not has_user_convergence) or is_converged(out);
293  auto const res = residual_convergence() <= 0e0 or residual_norm < residual_convergence();
294  auto const rel = relative_variation() <= 0e0 or relative_objective < relative_variation();
295 
296  converged = user and res and rel;
297  if(converged) {
298  SOPT_MEDIUM_LOG(" - [PD] converged in {} of {} iterations", niters, itermax());
299  break;
300  }
301  }
302  // check function exists, otherwise, don't know if convergence is meaningful
303  if(not converged)
304  SOPT_ERROR(" - [PD] did not converge within {} iterations", itermax());
305 
306  return {niters, converged, std::move(residual)};
307 }
