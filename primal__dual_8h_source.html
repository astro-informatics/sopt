282 template <typename SCALAR>
283 void PrimalDual<SCALAR>::iteration_step(t_Vector &out, t_Vector &out_hold, t_Vector &u,
284  t_Vector &u_hold, t_Vector &v, t_Vector &v_hold,
285  t_Vector &residual, t_Vector &q, t_Vector &r,
286  bool &random_measurement_update,
287  bool &random_wavelet_update, t_Vector &u_update,
288  t_Vector &v_update) const {
289  // dual calculations for measurements
290  if (random_measurement_update) {
291  g_proximal(v_hold, rho(), v + residual);
292  v_hold = v + residual - v_hold;
293  v = v + update_scale() * (v_hold - v);
294  v_update = static_cast<t_Vector>(Phi().adjoint() * v);
295  }
296  // dual calculations for wavelet
297  if (random_wavelet_update) {
298  q = static_cast<t_Vector>(Psi().adjoint() * out_hold) * sigma();
299  f_proximal(u_hold, gamma(), (u + q));
300  u_hold = u + q - u_hold;
301  u = u + update_scale() * (u_hold - u);
302  u_update = static_cast<t_Vector>(Psi() * u);
303  }
304  // primal calculations
305  r = out;
306 #ifdef SOPT_MPI
307  if (v_all_sum_all_comm().size() > 0 and u_all_sum_all_comm().size() > 0)
308  constraint()(
309  out_hold,
310  r - tau() * (u_all_sum_all_comm().all_sum_all(static_cast<const t_Vector>(u_update)) +
311  v_all_sum_all_comm().all_sum_all(static_cast<const t_Vector>(v_update))));
312  else
313 #endif
314  constraint()(out_hold, r - tau() * (u_update + v_update));
315  out = r + update_scale() * (out_hold - r);
316  out_hold = 2 * out_hold - r;
317  random_measurement_update = random_measurement_updater_();
318  random_wavelet_update = random_wavelet_updater_();
319  // update residual
320  if (random_measurement_update)
321  residual = static_cast<t_Vector>(Phi() * out_hold) * xi() - target();
322 }

324 template <typename SCALAR>
325 typename PrimalDual<SCALAR>::Diagnostic PrimalDual<SCALAR>::operator()(
326  t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
327  SOPT_HIGH_LOG("Performing Primal Dual");
328  sanity_check(x_guess, res_guess);
329  bool random_measurement_update = random_measurement_updater_();
330  bool random_wavelet_update = random_wavelet_updater_();
331  t_Vector residual = res_guess;
332  out = x_guess;
333  t_Vector out_hold = x_guess;
334  t_Vector r = x_guess;
335  t_Vector v = residual;
336  t_Vector v_hold = residual;
337  t_Vector v_update = x_guess;
338  t_Vector u = Psi().adjoint() * out;
339  t_Vector u_hold = u;
340  t_Vector u_update = out;
341  t_Vector q = u;
342 
343  t_uint niters(0);
344  bool converged = false;
345  for (; (not converged) && (niters < itermax()); ++niters) {
346  SOPT_LOW_LOG(" - [Primal Dual] Iteration {}/{}", niters, itermax());
347  iteration_step(out, out_hold, u, u_hold, v, v_hold, residual, q, r, random_measurement_update,
348  random_wavelet_update, u_update, v_update);
349  SOPT_LOW_LOG(" - [Primal Dual] Sum of residuals: {}",
350  static_cast<t_Vector>(residual).array().abs().sum());
351  converged = is_converged(out, residual);
352  }
353 
354  if (converged) {
355  SOPT_MEDIUM_LOG(" - [Primal Dual] converged in {} of {} iterations", niters, itermax());
356  } else if (static_cast<bool>(is_converged())) {
357  // not meaningful if not convergence function
358  SOPT_ERROR(" - [Primal Dual] did not converge within {} iterations", itermax());
359  }
360  return {niters, converged, std::move(residual)};
361 }
