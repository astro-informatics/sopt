281 template <class SCALAR>
282 void PrimalDual<SCALAR>::iteration_step(t_Vector &out, t_Vector &out_hold, t_Vector &u,
283  t_Vector &u_hold, t_Vector &v, t_Vector &v_hold,
284  t_Vector &residual, t_Vector &q, t_Vector &r,
285  bool &random_measurement_update,
286  bool &random_wavelet_update, t_Vector &u_update,
287  t_Vector &v_update) const {
288  // dual calculations for measurements
289  if (random_measurement_update) {
290  g_proximal(v_hold, rho(), v + residual);
291  v_hold = v + residual - v_hold;
292  v = v + update_scale() * (v_hold - v);
293  v_update = static_cast<t_Vector>(Phi().adjoint() * v);
294  }
295  // dual calculations for wavelet
296  if (random_wavelet_update) {
297  q = static_cast<t_Vector>(Psi().adjoint() * out_hold) * sigma();
298  f_proximal(u_hold, gamma(), (u + q));
299  u_hold = u + q - u_hold;
300  u = u + update_scale() * (u_hold - u);
301  u_update = static_cast<t_Vector>(Psi() * u);
302  }
303  // primal calculations
304  r = out;
305 #ifdef SOPT_MPI
306  if (v_all_sum_all_comm().size() > 0 and u_all_sum_all_comm().size() > 0)
307  constraint()(
308  out_hold,
309  r - tau() * (u_all_sum_all_comm().all_sum_all(static_cast<const t_Vector>(u_update)) +
310  v_all_sum_all_comm().all_sum_all(static_cast<const t_Vector>(v_update))));
311  else
312 #endif
313  constraint()(out_hold, r - tau() * (u_update + v_update));
314  out = r + update_scale() * (out_hold - r);
315  out_hold = 2 * out_hold - r;
316  random_measurement_update = random_measurement_updater_();
317  random_wavelet_update = random_wavelet_updater_();
318  // update residual
319  if (random_measurement_update)
320  residual = static_cast<t_Vector>(Phi() * out_hold) * xi() - target();
321 }

323 template <class SCALAR>
324 typename PrimalDual<SCALAR>::Diagnostic PrimalDual<SCALAR>::operator()(
325  t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
326  SOPT_HIGH_LOG("Performing Primal Dual");
327  sanity_check(x_guess, res_guess);
328  bool random_measurement_update = random_measurement_updater_();
329  bool random_wavelet_update = random_wavelet_updater_();
330  t_Vector residual = res_guess;
331  out = x_guess;
332  t_Vector out_hold = x_guess;
333  t_Vector r = x_guess;
334  t_Vector v = residual;
335  t_Vector v_hold = residual;
336  t_Vector v_update = x_guess;
337  t_Vector u = Psi().adjoint() * out;
338  t_Vector u_hold = u;
339  t_Vector u_update = out;
340  t_Vector q = u;
341 
342  t_uint niters(0);
343  bool converged = false;
344  for (; (not converged) && (niters < itermax()); ++niters) {
345  SOPT_LOW_LOG(" - [Primal Dual] Iteration {}/{}", niters, itermax());
346  iteration_step(out, out_hold, u, u_hold, v, v_hold, residual, q, r, random_measurement_update,
347  random_wavelet_update, u_update, v_update);
348  SOPT_LOW_LOG(" - [Primal Dual] Sum of residuals: {}",
349  static_cast<t_Vector>(residual).array().abs().sum());
350  converged = is_converged(out, residual);
351  }
352 
353  if (converged) {
354  SOPT_MEDIUM_LOG(" - [Primal Dual] converged in {} of {} iterations", niters, itermax());
355  } else if (static_cast<bool>(is_converged())) {
356  // not meaningful if not convergence function
357  SOPT_ERROR(" - [Primal Dual] did not converge within {} iterations", itermax());
358  }
359  return {niters, converged, std::move(residual)};
360 }
