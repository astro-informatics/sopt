276 template <class SCALAR>
277 void PrimalDual<SCALAR>::iteration_step(t_Vector &out, t_Vector &out_hold, t_Vector &u,
278  t_Vector &u_hold, t_Vector &v, t_Vector &v_hold,
279  t_Vector &residual, t_Vector &q, t_Vector &r,
280  bool &random_measurement_update,
281  bool &random_wavelet_update, t_Vector &u_update,
282  t_Vector &v_update) const {
283  // dual calculations for measurements
284  if (random_measurement_update) {
285  g_proximal(v_hold, rho(), v + residual);
286  v_hold = v + residual - v_hold;
287  v = v + update_scale() * (v_hold - v);
288  v_update = Phi().adjoint() * v;
289  }
290  // dual calculations for wavelet
291  if (random_wavelet_update) {
292  q = (Psi().adjoint() * out_hold) * sigma();
293  f_proximal(u_hold, gamma(), (u + q));
294  u_hold = u + q - u_hold;
295  u = u + update_scale() * (u_hold - u);
296  u_update = Psi() * u;
297  }
298  // primal calculations
299  r = out;
300 #ifdef SOPT_MPI
301  if (v_all_sum_all_comm().size() > 0 and u_all_sum_all_comm().size() > 0)
302  constraint()(
303  out_hold,
304  r - tau() * (u_all_sum_all_comm().all_sum_all(static_cast<const t_Vector>(u_update)) +
305  v_all_sum_all_comm().all_sum_all(static_cast<const t_Vector>(v_update))));
306  else
307 #endif
308  constraint()(out_hold, r - tau() * (u_update + v_update));
309  out = r + update_scale() * (out_hold - r);
310  out_hold = 2 * out_hold - r;
311  random_measurement_update = random_measurement_updater_();
312  random_wavelet_update = random_wavelet_updater_();
313  // update residual
314  if (random_measurement_update) residual = ((Phi() * out_hold) * xi() - target());
315 }

317 template <class SCALAR>
318 typename PrimalDual<SCALAR>::Diagnostic PrimalDual<SCALAR>::operator()(
319  t_Vector &out, t_Vector const &x_guess, t_Vector const &res_guess) const {
320  SOPT_HIGH_LOG("Performing Primal Dual");
321  sanity_check(x_guess, res_guess);
322  bool random_measurement_update = random_measurement_updater_();
323  bool random_wavelet_update = random_wavelet_updater_();
324  t_Vector residual = res_guess;
325  out = x_guess;
326  t_Vector out_hold = x_guess;
327  t_Vector r = x_guess;
328  t_Vector v = residual;
329  t_Vector v_hold = residual;
330  t_Vector v_update = x_guess;
331  t_Vector u = Psi().adjoint() * out;
332  t_Vector u_hold = u;
333  t_Vector u_update = out;
334  t_Vector q = u;
335 
336  t_uint niters(0);
337  bool converged = false;
338  for (; (not converged) && (niters < itermax()); ++niters) {
339  SOPT_LOW_LOG(" - [Primal Dual] Iteration {}/{}", niters, itermax());
340  iteration_step(out, out_hold, u, u_hold, v, v_hold, residual, q, r, random_measurement_update,
341  random_wavelet_update, u_update, v_update);
342  SOPT_LOW_LOG(" - [Primal Dual] Sum of residuals: {}", residual.array().abs().sum());
343  converged = is_converged(out, residual);
344  }
345 
346  if (converged) {
347  SOPT_MEDIUM_LOG(" - [Primal Dual] converged in {} of {} iterations", niters, itermax());
348  } else if (static_cast<bool>(is_converged())) {
349  // not meaningful if not convergence function
350  SOPT_ERROR(" - [Primal Dual] did not converge within {} iterations", itermax());
351  }
352  return {niters, converged, std::move(residual)};
353 }
